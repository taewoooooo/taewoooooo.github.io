迭代器Iterator

任何数据结构只要部署Iterator接口就可以使用迭代器遍历

用于ES6创建的for...of循环

原生具备iterator接口的数据： Array Arguments Set Map String TypedArray NodeList

工作原理：

  1. 创建一个指针对象指向当前数据结构的起始位置
  2. 第一次调用对象的next方法，指针自动指向数据结构的第一个成员
  3. 之后调用next指针会一直往后移，直到指向最后一个成员
  4. 每次调用next方法返回一个包含value和done属性的对象
const arr = [1, 2, 3, 4];
//for...of遍历
for(let a of arr){
  console.log(a); //输出1， 2， 3， 4
}
//手动迭代返回对象
let iterator = arr[Symbol.iterator]();
console.log(iterator.next()); //1
console.log(iterator.next()); //2
console.log(iterator.next()); //3
console.log(iterator.next()); //4

应用- 自定义遍历数据

const obj = {
  name: "qty",
  age: 18,
  arr: [1, 2, 3, 4],
  //添加迭代器
  [Symbol.iterator](){
    let index = 0;
    let _this = this;
    return {
      next: function(){ //里面不能直接用this，因为对象里面的this指向对象
        if(index < _this.arr.length){
          return {value: _this.arr[index++], done: false};
        }else{
          return {value: undefined, done: true};
        }
      }
    };
  }
};
//要求for...of每次遍历arr中的元素
for(let a of obj){
  console.log(a);
}

生成器

是一个函数，一种异步编程解决方案

function * gen(){
  //三个yield将代码块分成四部分，用next依次执行
  console.log("第一段");
  let one = yield "aa";
  console.log(one);
  let two = yield "bb";
  console.log(two);
  let three = yield "cc";
  console.log(three);
}
let iterator = gen(); //返回的是迭代器实例对象，此时无法执行函数内的代码


//依次执行四部分代码片段，next里面可以传参数，用于设置前面一个yield的值
console.log(iterator.next());  //执行第一段代码 + 返回第一个yield后面的变量，并将其作为对象
console.log(iterator.next("ssss"));  //执行第二段代码 + 返回第二个yield后面的变量，并将其作为对象,one="ssss"
console.log(iterator.next("sss")); //two = "sss"
console.log(iterator.next());

//for...of遍历
for(let r of iterator){
  console(r); //返回yield中的变量做为对象，同时执行代码块语句
}

Promise

ES6引入的异步编程解决方案，是一个构造函数

实例化Promise对象

const p = new Promise(function(resolve, rejcet){
  setTimeout(function(){
      let data = "data";
    //成功时调用resolve
    resolve(data);
    //失败时调用reject
    reject(data);
  }, 1000);
});

//then的第一个参数是成功时执行的函数，对应构造函数中返回resolve()
//第二个是失败时执行的函数,对应构造函数中返回reject()
p.then(function(value){
  console.log(value);
}, funtion(reason){
  console.error(reason);
});

Promise读取文件

//1.引入fs模块（node.js）
const fs = require("fs");
//2.调用方法读取文件
fs.readFile("./1.txt", (err, data)=>{
  if(err)
    throw err;
  console.log(data.toString());
});


//Promise封装
const p = new Promise(function(resolve, reject){
  fs.readFile("./1.txt", (err, data)=>{
    if(err)
      reject(err);
    resolve(data);
  })
});
p.then(function(value){
  console.log(value);
},function(reason){
  console.error(reason);
});

Promise封装AJAX请求

const p = new Promise(function(resolve, reject){
  const xhr = new XMLHttpRequest();
  xhr.open("GET", "./1.php");
  xhr.send();
  xhr.onreadystatechange = function(){
    if(xhr.readyState == 4){
      if(xhr.status >= 200 && xhr.status < 300 || xhr.status == 304)
        resolve(xhr.responseText);
      else
        reject(xhr.status);
    }
  }
});

p.then(function(value){
  console.log(value);
}, function(reason){
  console.error(reason);
});
