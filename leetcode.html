455. 分发饼干

class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
				// i,j是工作指针，分别在g和s中遍历
        int i = 0, j = 0, count = 0;
        
        // 先对s和g进行排序
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());

        while(i<g.size() && j<s.size()){
            // 如果s[j]可以满足g[i]
            if(s[j] >= g[i]){
                i++;
                count++;
            }
            // 满不满足糖果的j都要往后移动
            j++;
        }
    return count;
    }
};

376. 摇摆序列

class Solution {
public:
    int wiggleMaxLength(vector<int>& nums) {
        // 贪心算法，只留上升子序列中的最大值和下降子序列的最小值
        // 使用状态机

        if(nums.size() < 2){
            return nums.size();
        }
        
        // 定义状态机状态量
        static const int BEGIN = 0;
        static const int UP = 1;
        static const int DOWN = 2;

        int max_length = 1, STATE = BEGIN;

        // 从数组的第二个数开始，判断前一个元素是否算在摇摆序列内
        for(int i=1; i<nums.size(); i++){
            switch(STATE){
                case BEGIN:
                    if(nums[i] < nums[i-1]){
                        // 初态变为降序，第一个元素算数
                        STATE = DOWN;
                        max_length++;
                    }else if(nums[i] > nums[i-1]){
                        // 初态变为升序，第一个元素算数
                        STATE = UP;
                        max_length++;
                    }else{
                        // 如果相等，则保持BEGIN状态，什么也不用做
                    }
                    break;
                case UP:
                    if(nums[i] < nums[i-1]){
                        // 由升序转变为降序，前一个元素算数
                        STATE = DOWN;
                        max_length++;
                    }
                    break;
                case DOWN:
                    if(nums[i] > nums[i-1]){
                        STATE = UP;
                        max_length++;
                    }
                    break;
            }
        }
        return max_length;

    }
};

22. 括号生成

class Solution {
public:
    vector<string> generateParenthesis(int n) {
        string item;
        vector<string> result;

        generate(item, result, n, n);
        return result;
    }

private: 
    // left和righe是左括号和右括号还剩多少可以用
    void generate(string item, vector<string>& result, int left, int right){
        // 递归出口
        if(left == 0 && right == 0){
            result.push_back(item);
            return;
        }
        // 剪枝
        if(left > 0){
            generate(item+'(', result, left-1, right);
        }
        if(left < right){
            generate(item+')', result, left, right-1);
        }
    }
};

class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {
				vector<vector<int>> result;
		        vector<int> item;
		        // 递归回溯是从 [1] 开始的，没有考虑空数组
		        result.push_back(item);
		        generate(0, nums, item, result);
		        return result;
		}
private:
    // 递归函数
    void generate(int i, vector<int>& nums, vector<int>& item,vector<vector<int>>& result){
        // 递归出口
        if(i >= nums.size()){
            return;
        }
        // 递归带当前i的子集
        item.push_back(nums[i]);
        result.push_back(item);
        generate(i+1, nums, item, result);
        // 回溯不带当前的i的子集
        item.pop_back();
        generate(i+1, nums, item, result);
		}
};

法二：位运算

class Solution {
public:
    vector<vector<int>> subsets(vector<int>& nums) {

        //位运算
        int all_set = 1 << nums.size();

        vector<vector<int>> result;
        for(int i=0; i<all_set; i++){
            vector<int> item;
            for(int j=0; j<nums.size(); j++){
                if(i&(1<<j)){
                    item.push_back(nums[j]);
                }
            }
            result.push_back(item);
        }
        return result;
}

90.子集II

class Solution {
public:
    vector<vector<int>> subsetsWithDup(vector<int>& nums) {
        // 对nums进行排序，防止出现[1,2,2]和[2,1,2]相同的情况
        sort(nums.begin(), nums.end());
        //使用set对结果去重
        set<vector<int>> res_set;
        vector<int> item;
        vector<vector<int>> result;
        
        result.push_back(item);
        generate(0, nums, item, result, res_set);
        return result;
    }

private:
    void generate(int i, vector<int>& nums, vector<int>& item, vector<vector<int>>& result, set<vector<int>>& res_set){
        // 递归出口
        if(i >= nums.size()){
            return;
        }
        //利用set去重
        item.push_back(nums[i]);
        if(res_set.find(item) == res_set.end()){
            result.push_back(item);
            res_set.insert(item);
        }

        generate(i+1, nums, item, result, res_set);
        item.pop_back();
        generate(i+1, nums, item, result, res_set);
    }
};

51.N皇后

class Solution {
public:
    vector<vector<string>> solveNQueens(int n) {
        // 存放最终的可行结果
        vector<vector<string>> result;
        // 存放每次摆放后的棋局可落子位置（0 1表示）
        vector<vector<int>> mark;
        // 存放某次摆放后的结果
        vector<string> location;
        for(int i=0; i<n; i++){
            vector<int> tmp_vec;
            for(int j=0; j<n; j++){
                tmp_vec.push_back(0);
            }
            mark.push_back(tmp_vec);


            location.push_back("");
            location[i].append(n, '.');
        }
        //以上是数据初始化


        generate(0, n, location, result, mark);
        return result;
    }
private:
    // 在(x,y)放下一个皇后对棋盘可用位置的影响(0为可用 1为不可用)
    void put_down_the_queen(int x, int y, vector<vector<int>>& mark){
        // 方向数组
        static const int dx[] = {-1, -1, 0, 1, 1, 1, 1, -1};
        static const int dy[] = {0, -1, -1, -1, 0, 1, 0, 1};

        // 在(x,y)放下皇后
        mark[x][y] = 1;
        // 影响了每个方向至多多少格子
        for(int i=1; i<mark.size(); i++){
            // 影响了八个方向
            for(int j=0; j<8; j++){
                int new_x =  x + dx[j]*i;
                int new_y = y + dy[j]*i;
                if(new_x>=0 && new_x<mark.size() && new_y>=0 && new_y<mark.size()){
                    // 更新棋盘
                    mark[new_x][new_y] = 1;
                }
            }
        }
    }
    // k是正在放置的第k行皇后
    void generate(int k, int n, vector<string>& location, vector<vector<string>>& result, vector<vector<int>>& mark){
        // 递归出口
        if(k == n){
            result.push_back(location);
            return;
        }
        // 扫描第k行所有列，寻找可以放皇后的位置
        for(int i=0; i<n; i++){
            if(mark[k][i] == 0){
                // 记录添加皇后前的mark，以备回溯之用
                vector<vector<int>> tmp_mark = mark;
                // 更新location
                location[k][i] = 'Q';
                // 在(k, i)位置放入皇后，计算mark变化
                put_down_the_queen(k, i, mark);
                // 递归下一行的皇后位置
                generate(k+1, n, location, result, mark);
                // 如果将来回溯到这里，使用tmp_mark还原mark，并还原location
                mark = tmp_mark;
                location[k][i] = '.';
            }
        }
    }

};

113. 路径总和 II

class Solution {
public:
    vector<vector<int>> pathSum(TreeNode* root, int targetSum) {
        vector<vector<int>> result;
        vector<int> path;
        int path_value = 0;
        preorder(root, path_value, targetSum, path, result);
        return result;
    }
private:
    // path_value需要带上引用符号，因为每次遍历都会更改这个值
    void preorder(TreeNode* node, int &path_value, int targetSum, vector<int> &path, vector<vector<int>> &result){
        // 递归出口
        if(node == nullptr){
            return;
        }

        // 在遍历前需要做的事情
        path_value += node->val;
        path.push_back(node->val);
        // 符合条件，则添加到result
        if(node->left == nullptr && node->right == nullptr && path_value == targetSum){
            result.push_back(path);
        }
        preorder(node->left, path_value, targetSum, path, result);
        // 遍历完左子树需要做的事情

        preorder(node->right, path_value, targetSum, path, result);
        // 遍历完左右子树后需要做的事情，回溯
        path_value -= node->val;
        path.pop_back();
    }
};

236. 二叉树的最近公共祖先

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        // 存储节点路径的栈
        vector<TreeNode*> path;
        // 存储p和q对应的路径数组
        vector<TreeNode*> node_p_path;
        vector<TreeNode*> node_q_path;
        int finish = 0;
        // 搜索p的路径
        preorder(root, p, path, node_p_path, finish);
        // 初始化finish和path，为q的搜索做准备
        finish = 0;
        path.clear();
        preorder(root, q, path, node_q_path, finish);
        
        // 比较两个数组，找相同的最大长度
        int path_len = node_p_path.size() > node_q_path.size() ? node_q_path.size() : node_p_path.size();
        // 同时遍历两个数组
        TreeNode * result;
        for(int i=0; i<path_len; i++){
            if(node_p_path[i] == node_q_path[i]){
                // 不断更新result，直到两数组遇到不同元素，此时的result为最后结果
                result = node_p_path[i];
            }
        }
        return result;
    }
private:
    // node是正在遍历的节点， search是需要搜索的节点， path是遍历到当前节点的路径栈
    // result是搜索到search时的路径， finish标记是否找到search的位置
    void preorder(TreeNode *node, TreeNode *search, vector<TreeNode*> &path, vector<TreeNode*> &result, int &finish){
        // 递归出口
        if(node == NULL || finish == 1){
            return;
        }
        // 遍历遍历前的准备
        path.push_back(node);
        if(node == search){
            finish = 1;
            result = path;
        }

        preorder(node->left, search, path, result, finish);

        preorder(node->right, search, path, result, finish);

        //回溯，遍历完node节点的左右子树后，将node弹出栈
        path.pop_back();
    }
};

阿里笔试题

输入n和k，求从0到n中所有不相等的(x,y)的个数，其中x%y=k. (5,2 ⇒ 4. 7,3⇒5)

int res = 0, n, k;
cin>>n>>k;
for(int y = k+1; y<n; y++){
	tmp = y/k;
	if(n%y >= k)
		tmp++;
	res += tmp;
}
cout<<res<<endl;
return 0;
